def turns
        while winner?(@gameboard.board) == false
                puts "\n\nIt is now #{@current_player.name}'s turn."
                puts @gameboard.draw_board
                @gameboard.update_board(@current_player.side)
                puts @gameboard.draw_board
                switch_player
        end   
end
